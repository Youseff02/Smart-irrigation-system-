/* ESP8266 Multi-Sensor Live Dashboard — Sensors + Pump + LCD I2C
   ══════════════════════════════════════════════════════════════════
   Sensors:
     • Soil Moisture  → A0         (analog, 0-1024 raw)
     • Water Level    → D7 GPIO13  (digital, holds DETECTED for 100 s)
     • DHT11          → D4 GPIO2   (temperature + humidity)

   Actuator:
     • Pump relay     → D1 GPIO5
       Most relay modules are ACTIVE-LOW — set RELAY_ACTIVE_HIGH = false (default).
       If pump logic is inverted, set RELAY_ACTIVE_HIGH = true.

   Display:
     • LCD 16×2 I2C   → SDA=D2 GPIO4 / SCL=D3 GPIO0  (address 0x27 or 0x3F)

   WiFi: STEP  /  192.168.1.221
   ══════════════════════════════════════════════════════════════════
   Libraries (Arduino Library Manager):
     • DHT sensor library        — Adafruit
     • Adafruit Unified Sensor   — Adafruit
     • LiquidCrystal_I2C         — Frank de Brabander
     • Wire                      — built-in
*/

#include <ESP8266WiFi.h>
#include <DHT.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>

// ── WiFi ──────────────────────────────────────────────────────────────────────
const char* ssid     = "STEP";
const char* password = "Step@Bns";

// ── Pins ──────────────────────────────────────────────────────────────────────
const int waterPin = 13;   // D7 — digital water level sensor (DO pin)
const int pumpPin  =  5;   // D1 — pump relay signal pin
#define   DHTPIN   2        // D4 — DHT11 data
#define   DHTTYPE  DHT11

// ── Relay polarity ────────────────────────────────────────────────────────────
// Most cheap relay modules are ACTIVE-LOW:
//   pin LOW  → relay ON  (pump runs)
//   pin HIGH → relay OFF (pump stops)
// If your pump behaves the OPPOSITE way, change this to true.
const bool RELAY_ACTIVE_HIGH = false;

// ── LCD ───────────────────────────────────────────────────────────────────────
// Try address 0x27 first; if LCD stays blank, change to 0x3F
LiquidCrystal_I2C lcd(0x27, 16, 2);

// ── Objects ───────────────────────────────────────────────────────────────────
WiFiServer server(80);
DHT dht(DHTPIN, DHTTYPE);

// ── Sensor readings ───────────────────────────────────────────────────────────
int   moistureRaw   = 0;
float moistureVolts = 0.0;
bool  waterPresent  = false;
float temperature   = 0.0;
float humidity      = 0.0;

// ── Pump state ────────────────────────────────────────────────────────────────
bool pumpOn = false;

// ── Water hold timer ──────────────────────────────────────────────────────────
unsigned long waterLastDetectedMs = 0;
const unsigned long WATER_HOLD_MS = 100UL * 1000UL;  // 100 seconds in ms

// ── Timing ────────────────────────────────────────────────────────────────────
unsigned long lastReadMs = 0;
unsigned long lastLcdMs  = 0;
int           lcdPage    = 0;   // cycles 0-2

// ─────────────────────────────────────────────────────────────────────────────
// applyPump — drives the relay pin correctly for active-HIGH or active-LOW boards
// ─────────────────────────────────────────────────────────────────────────────
void applyPump(bool state) {
  pumpOn = state;
  // active-HIGH: HIGH=ON / LOW=OFF
  // active-LOW:  LOW=ON  / HIGH=OFF
  digitalWrite(pumpPin, (RELAY_ACTIVE_HIGH ? pumpOn : !pumpOn) ? HIGH : LOW);
}

// ─────────────────────────────────────────────────────────────────────────────
void readAllSensors() {
  // ── Soil moisture (A0) — average 8 samples ───────────────────────────────
  long sum = 0;
  for (int i = 0; i < 8; i++) { sum += analogRead(A0); delay(3); }
  moistureRaw   = (int)(sum / 8);
  moistureVolts = (moistureRaw * 3.08f) / 1024.0f;

  // ── Water level (D7) — majority vote of 3 reads to debounce ─────────────
  int w1 = digitalRead(waterPin); delay(5);
  int w2 = digitalRead(waterPin); delay(5);
  int w3 = digitalRead(waterPin);
  bool sensorHigh = ((w1 + w2 + w3) >= 2);

  if (sensorHigh) {
    waterLastDetectedMs = millis();
    waterPresent = true;
  } else {
    if (waterPresent && (millis() - waterLastDetectedMs >= WATER_HOLD_MS)) {
      waterPresent = false;
    }
  }

  // ── DHT11 (D4) — up to 3 attempts ────────────────────────────────────────
  for (int attempt = 0; attempt < 3; attempt++) {
    float h = dht.readHumidity();
    float t = dht.readTemperature();
    if (!isnan(h) && !isnan(t)) {
      humidity    = h;
      temperature = t;
      break;
    }
    delay(100);  // 100ms بدل 300ms — كافية للـ DHT11 وبتقلل الـ blocking
  }
}

// ─────────────────────────────────────────────────────────────────────────────
void printSerial() {
  unsigned long holdRemain = 0;
  if (waterPresent && waterLastDetectedMs > 0) {
    unsigned long elapsed = millis() - waterLastDetectedMs;
    holdRemain = (elapsed < WATER_HOLD_MS) ? (WATER_HOLD_MS - elapsed) / 1000 : 0;
  }
  Serial.println(F("─────────────────────────────────────────"));
  Serial.print(F("Moisture  Raw : ")); Serial.print(moistureRaw);
  Serial.print(F("  Volts: "));        Serial.print(moistureVolts, 2); Serial.println(F(" V"));
  Serial.print(F("Water Level   : "));
  Serial.print(waterPresent ? F("WATER DETECTED") : F("NO WATER"));
  if (waterPresent && holdRemain > 0) {
    Serial.print(F("  (clears in ")); Serial.print(holdRemain); Serial.print(F(" s)"));
  }
  Serial.println();
  Serial.print(F("Temperature   : ")); Serial.print(temperature, 1); Serial.println(F(" °C"));
  Serial.print(F("Humidity      : ")); Serial.print(humidity, 1);    Serial.println(F(" %"));
  Serial.print(F("Pump          : ")); Serial.println(pumpOn ? F("ON") : F("OFF"));
}

// ─────────────────────────────────────────────────────────────────────────────
// LCD — rotates through 3 pages every 3 s
// Page 0: Moisture raw + volts
// Page 1: Water level + temperature
// Page 2: Humidity + pump state
// ─────────────────────────────────────────────────────────────────────────────
void updateLCD() {
  lcd.clear();
  switch (lcdPage) {
    case 0:
      lcd.setCursor(0, 0); lcd.print("Moisture:");
      lcd.setCursor(9, 0); lcd.print(moistureRaw);
      lcd.setCursor(0, 1); lcd.print("Volts: ");
      lcd.print(moistureVolts, 2); lcd.print("V");
      break;
    case 1:
      lcd.setCursor(0, 0); lcd.print("Water:");
      lcd.setCursor(7, 0); lcd.print(waterPresent ? "DETECTED" : "NONE    ");
      lcd.setCursor(0, 1); lcd.print("Temp: ");
      lcd.print(temperature, 1); lcd.print((char)0xDF); lcd.print("C");
      break;
    case 2:
      lcd.setCursor(0, 0); lcd.print("Humidity: ");
      lcd.print(humidity, 0); lcd.print("%");
      lcd.setCursor(0, 1); lcd.print("Pump: ");
      lcd.print(pumpOn ? "ON          " : "OFF         ");
      break;
  }
  lcdPage = (lcdPage + 1) % 3;
}

// ─────────────────────────────────────────────────────────────────────────────
String buildJSON() {
  unsigned long holdRemain = 0;
  if (waterPresent && waterLastDetectedMs > 0) {
    unsigned long elapsed = millis() - waterLastDetectedMs;
    holdRemain = (elapsed < WATER_HOLD_MS) ? (WATER_HOLD_MS - elapsed) / 1000 : 0;
  }
  String j = "{";
  j += "\"moistureRaw\":"  + String(moistureRaw)          + ",";
  j += "\"moistureV\":"    + String(moistureVolts, 2)     + ",";
  j += "\"water\":"        + String(waterPresent ? 1 : 0) + ",";
  j += "\"waterHold\":"    + String(holdRemain)           + ",";
  j += "\"temp\":"         + String(temperature, 1)       + ",";
  j += "\"humidity\":"     + String(humidity, 1)          + ",";
  j += "\"pump\":"         + String(pumpOn ? 1 : 0);
  j += "}";
  return j;
}

// ─────────────────────────────────────────────────────────────────────────────
void setup() {
  Serial.begin(115200);
  delay(10);

  // ── Pump pin — set level BEFORE pinMode to avoid boot glitch ─────────────
  // Write the "OFF" level first so the relay doesn't fire during boot
  digitalWrite(pumpPin, RELAY_ACTIVE_HIGH ? LOW : HIGH);  // OFF state
  pinMode(pumpPin, OUTPUT);
  digitalWrite(pumpPin, RELAY_ACTIVE_HIGH ? LOW : HIGH);  // reinforce OFF
  pumpOn = false;

  Serial.println(F("\n[RELAY TEST] ON for 1 s..."));
  applyPump(true);
  delay(1000);
  applyPump(false);
  Serial.println(F("[RELAY TEST] OFF — if relay clicked twice, wiring is good."));

  pinMode(waterPin, INPUT);
  dht.begin();

  // ── LCD ───────────────────────────────────────────────────────────────────
  Wire.begin(4, 0);       // SDA=D2(GPIO4), SCL=D3(GPIO0)
  lcd.init();
  lcd.backlight();
  lcd.setCursor(0, 0); lcd.print("  Plant Monitor ");
  lcd.setCursor(0, 1); lcd.print("  Connecting... ");
  delay(1000);

  // ── WiFi ──────────────────────────────────────────────────────────────────
  Serial.println();
  Serial.print(F("Connecting to ")); Serial.println(ssid);
  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print(F(".")); }

  Serial.println(F("\nWiFi connected"));
  server.begin();
  Serial.println(F("Server started"));
  Serial.print(F("Dashboard: http://")); Serial.println(WiFi.localIP());

  lcd.clear();
  lcd.setCursor(0, 0); lcd.print("WiFi Connected!");
  lcd.setCursor(0, 1); lcd.print(WiFi.localIP());
  delay(2000);

  readAllSensors();
  printSerial();
  updateLCD();
  lastReadMs = millis();
  lastLcdMs  = millis();
}

// ─────────────────────────────────────────────────────────────────────────────
void loop() {

  // ── Read sensors every 2 s ───────────────────────────────────────────────
  if (millis() - lastReadMs >= 2000) {
    lastReadMs = millis();
    readAllSensors();
    printSerial();
  }

  // ── Update LCD every 3 s (cycle pages) ───────────────────────────────────
  if (millis() - lastLcdMs >= 3000) {
    lastLcdMs = millis();
    updateLCD();
  }

  // ── Handle HTTP request ──────────────────────────────────────────────────
  WiFiClient client = server.available();
  if (!client) return;

  // ── Wait for the client to send data (timeout = 2 s) ─────────────────────
  // Prevents processing an empty request if the client connects but hasn't
  // finished sending headers yet.
  unsigned long waitStart = millis();
  while (client.available() == 0) {
    if (millis() - waitStart > 2000) {
      client.stop();
      return;
    }
  }

  String request = client.readStringUntil('\r');
  client.flush();

  if (request.indexOf(F("GET /cmd")) >= 0) {
    if (request.indexOf(F("pump_on"))  >= 0) applyPump(true);
    if (request.indexOf(F("pump_off")) >= 0) applyPump(false);

    readAllSensors();
    printSerial();
    lastReadMs = millis();

    // ── Send JSON with Content-Length so the browser knows when response ends
    String json = buildJSON();
    client.print(F("HTTP/1.1 200 OK\r\nContent-Type: application/json\r\n"
                   "Access-Control-Allow-Origin: *\r\nConnection: close\r\n"
                   "Content-Length: "));
    client.print(json.length());
    client.print(F("\r\n\r\n"));
    client.print(json);

  } else if (request.indexOf(F("GET /data")) >= 0) {
    readAllSensors();
    printSerial();
    lastReadMs = millis();

    // ── Send JSON with Content-Length so the browser knows when response ends
    String json = buildJSON();
    client.print(F("HTTP/1.1 200 OK\r\nContent-Type: application/json\r\n"
                   "Access-Control-Allow-Origin: *\r\nConnection: close\r\n"
                   "Content-Length: "));
    client.print(json.length());
    client.print(F("\r\n\r\n"));
    client.print(json);

  } else {
    // ── Serve the dashboard HTML page ────────────────────────────────────────
    String page = buildPage();
    client.print(F("HTTP/1.1 200 OK\r\nContent-Type: text/html\r\nConnection: close\r\n"
                   "Content-Length: "));
    client.print(page.length());
    client.print(F("\r\n\r\n"));
    client.print(page);
  }

  // ── Flush the TX buffer then cleanly close the connection ─────────────────
  // The small delay gives the ESP8266 TCP stack time to push remaining bytes
  // before the socket is torn down — without it the last chunk can be lost.
  delay(5);
  client.stop();
}
